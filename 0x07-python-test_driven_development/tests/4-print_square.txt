def test_print_square():
    # Test with size = 0
    assert print_square(0) == None

    # Test with size = 1
    assert print_square(1) == print("#")

    # Test with size = 2
    assert print_square(2) == print("##\n##")

    # Test with size = 5
    assert print_square(5) == print("#####\n#####\n#####\n#####\n#####")

    # Test with size = -1
    try:
        print_square(-1)
    except ValueError as e:
        assert str(e) == "size must be >= 0"

    # Test with size = 3.5
    try:
        print_square(3.5)
    except TypeError as e:
        assert str(e) == "size must be an integer"

    # Test with size = "foo"
    try:
        print_square("foo")
    except TypeError as e:
        assert str(e) == "size must be an integer"
