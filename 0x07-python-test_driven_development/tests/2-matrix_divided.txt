def test_matrix_divided():
    # Test dividing a matrix with integers
    matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
    assert matrix_divided(matrix, 2) == [[0.5, 1.0, 1.5], [2.0, 2.5, 3.0], [3.5, 4.0, 4.5]]
    # Test dividing a matrix with floats
    matrix = [[1.0, 2.0], [3.0, 4.0]]
    assert matrix_divided(matrix, 0.5) == [[2.0, 4.0], [6.0, 8.0]]
    # Test dividing a matrix with negative numbers
    matrix = [[-1, -2, -3], [-4, -5, -6]]
    assert matrix_divided(matrix, -2) == [[0.5, 1.0, 1.5], [2.0, 2.5, 3.0]]
    # Test dividing a matrix with zeros
    matrix = [[0, 0], [0, 0]]
    try:
        matrix_divided(matrix, 2)
    except ZeroDivisionError as e:
        assert str(e) == "division by zero"
    # Test dividing a matrix with a non-number divisor
    matrix = [[1, 2], [3, 4]]
    try:
        matrix_divided(matrix, "a")
    except TypeError as e:
        assert str(e) == "div must be a number"
    # Test dividing a non-matrix
    matrix = [1, 2, 3]
